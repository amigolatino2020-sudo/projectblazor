### Task Filter Component
```csharp
// Copilot generated filter component
<select @bind="selectedFilter" @onchange="ApplyFilter">
    <option value="all">All Tasks</option>
    <option value="completed">Completed</option>
    <option value="pending">Pending</option>
    <option value="overdue">Overdue</option>
</select>

@code {
    private string selectedFilter = "all";
    
    private async Task ApplyFilter()
    {
        var allTasks = await TaskService.GetAllTasksAsync();
        tasks = selectedFilter switch
        {
            "completed" => allTasks.Where(t => t.IsComplete).ToList(),
            "pending" => allTasks.Where(t => !t.IsComplete).ToList(),
            "overdue" => allTasks.Where(t => !t.IsComplete && t.DueDate < DateTime.Now).ToList(),
            _ => allTasks
        };
        StateHasChanged();
    }
}
